* Motivation

When an LSP server is configured but not found on PATH, diagnostics silently
don't work. Users may not realize they need to install a server. Auto-install
(or prompt-to-install) makes the experience smoother.

* Scope

When an LSP server is needed but missing, offer to install it automatically
using the appropriate package manager. Start with servers that have well-known
install commands.

* Current State

- ~is_command_available()~ (kodo-lsp/config.rs:45-52) checks if a command exists.
- ~LspManager~ silently skips servers that aren't on PATH.
- No install logic or user prompting.

#+begin_src rust
use std::collections::HashMap;

/// An install recipe for an LSP server.
#[derive(Debug, Clone)]
pub struct InstallRecipe {
    /// Human-readable server name.
    pub name: String,
    /// Install command to run (e.g., "npm install -g typescript-language-server").
    pub command: String,
    /// Arguments to the install command.
    pub args: Vec<String>,
    /// Optional: prerequisite check (e.g., "npm" must exist).
    pub requires: Option<String>,
}

/// Get the install recipe for a known LSP server.
/// Returns None for unknown servers.
pub fn install_recipe(server_name: &str) -> Option<InstallRecipe> {
    match server_name {
        "rust-analyzer" => Some(InstallRecipe {
            name: "rust-analyzer".into(),
            command: "rustup".into(),
            args: vec!["component".into(), "add".into(), "rust-analyzer".into()],
            requires: Some("rustup".into()),
        }),
        "gopls" => Some(InstallRecipe {
            name: "gopls".into(),
            command: "go".into(),
            args: vec!["install".into(), "golang.org/x/tools/gopls@latest".into()],
            requires: Some("go".into()),
        }),
        "typescript-language-server" => Some(InstallRecipe {
            name: "typescript-language-server".into(),
            command: "npm".into(),
            args: vec![
                "install".into(),
                "-g".into(),
                "typescript-language-server".into(),
                "typescript".into(),
            ],
            requires: Some("npm".into()),
        }),
        "pyright" | "pyright-langserver" => Some(InstallRecipe {
            name: "pyright".into(),
            command: "pip".into(),
            args: vec!["install".into(), "pyright".into()],
            requires: Some("pip".into()),
        }),
        _ => None,
    }
}

/// Get all known install recipes.
pub fn all_recipes() -> Vec<InstallRecipe> {
    ["rust-analyzer", "gopls", "typescript-language-server", "pyright"]
        .iter()
        .filter_map(|name| install_recipe(name))
        .collect()
}

#+begin_src rust
use anyhow::{Result, bail};
use tracing::{debug, info, warn};

use crate::config::is_command_available;

/// Attempt to install an LSP server using its recipe.
///
/// Returns Ok(true) if installed successfully, Ok(false) if prerequisites
/// are missing, or Err if the install command failed.
pub async fn install_server(recipe: &InstallRecipe) -> Result<bool> {
    // Check prerequisite
    if let Some(ref req) = recipe.requires {
        if !is_command_available(req) {
            warn!(
                server = %recipe.name,
                requires = %req,
                "cannot auto-install: prerequisite not found"
            );
            return Ok(false);
        }
    }

    info!(server = %recipe.name, command = %recipe.command, "installing LSP server");

    let output = tokio::process::Command::new(&recipe.command)
        .args(&recipe.args)
        .stdout(std::process::Stdio::piped())
        .stderr(std::process::Stdio::piped())
        .output()
        .await?;

    if output.status.success() {
        info!(server = %recipe.name, "LSP server installed successfully");
        Ok(true)
    } else {
        let stderr = String::from_utf8_lossy(&output.stderr);
        warn!(
            server = %recipe.name,
            stderr = %stderr,
            "LSP server install failed"
        );
        bail!("install failed for {}: {}", recipe.name, stderr)
    }
}

#+begin_src rust
use crate::installer;

impl LspManager {
    /// Check if a server is available; if not, try to auto-install it.
    /// Returns true if the server is now available (was already present or installed).
    pub async fn ensure_server_available(&mut self, server_name: &str) -> bool {
        // Already available?
        if is_command_available(server_name) {
            return true;
        }

        // Try auto-install
        if let Some(recipe) = installer::install_recipe(server_name) {
            match installer::install_server(&recipe).await {
                Ok(true) => {
                    tracing::info!(server = %server_name, "auto-installed LSP server");
                    true
                }
                Ok(false) => {
                    tracing::debug!(server = %server_name, "prerequisites missing for auto-install");
                    false
                }
                Err(e) => {
                    tracing::warn!(server = %server_name, error = %e, "auto-install failed");
                    false
                }
            }
        } else {
            tracing::debug!(server = %server_name, "no install recipe available");
            false
        }
    }
}
