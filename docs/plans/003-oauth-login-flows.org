* Motivation

Auth currently requires manually setting environment variables
(~ANTHROPIC_API_KEY~, ~OPENAI_API_KEY~, etc.). This is friction for new users
and doesn't support token refresh.

* Scope

Add OAuth2 Authorization Code + PKCE flow for providers that support it.
Start with a credential resolution chain (env var -> keychain -> prompt).

* Current State

- Providers use ~from_env()~ to read env vars (anthropic.rs:237, openai, gemini).
- ~kodo-store/auth.rs~ has ~AuthToken~ struct with ~token~, ~refresh_token~,
  ~expires_at~. Secrets go to OS keychain via ~keyring~ crate.
- ~auth_providers~ table in SQLite stores only provider name + expiry.
- No OAuth client, no browser flow, no token refresh.

use anyhow::Result;
use base64::Engine;
use base64::engine::general_purpose::URL_SAFE_NO_PAD;
use rand::Rng;
use sha2::{Sha256, Digest};

/// Generate a random PKCE code_verifier (43-128 chars, URL-safe).
pub fn generate_code_verifier() -> String {
    let mut rng = rand::thread_rng();
    let bytes: Vec<u8> = (0..32).map(|_| rng.gen()).collect();
    URL_SAFE_NO_PAD.encode(&bytes)
}

/// Derive the S256 code_challenge from a code_verifier.
pub fn code_challenge(verifier: &str) -> String {
    let hash = Sha256::digest(verifier.as_bytes());
    URL_SAFE_NO_PAD.encode(&hash)
}
#+end_src

/// OAuth provider configuration.
#[derive(Debug, Clone)]
pub struct OAuthConfig {
    pub provider: String,
    pub client_id: String,
    pub auth_url: String,
    pub token_url: String,
    pub scopes: Vec<String>,
}

/// Result of a successful OAuth flow.
#[derive(Debug)]
pub struct OAuthTokens {
    pub access_token: String,
    pub refresh_token: Option<String>,
    pub expires_in: Option<u64>,
}

/// Run the full OAuth Authorization Code + PKCE flow.
///
/// 1. Start a local HTTP server on a random port.
/// 2. Open the browser to the authorize URL.
/// 3. Wait for the callback with the auth code.
/// 4. Exchange the code for tokens.
/// 5. Return the tokens.
pub async fn run_oauth_flow(config: &OAuthConfig) -> Result<OAuthTokens> {
    let verifier = generate_code_verifier();
    let challenge = code_challenge(&verifier);

    // Bind to a random port on localhost
    let listener = tokio::net::TcpListener::bind("127.0.0.1:0").await?;
    let port = listener.local_addr()?.port();
    let redirect_uri = format!("http://127.0.0.1:{}/callback", port);

    // Build the authorize URL
    let mut auth_url = url::Url::parse(&config.auth_url)?;
    auth_url.query_pairs_mut()
        .append_pair("response_type", "code")
        .append_pair("client_id", &config.client_id)
        .append_pair("redirect_uri", &redirect_uri)
        .append_pair("code_challenge", &challenge)
        .append_pair("code_challenge_method", "S256")
        .append_pair("scope", &config.scopes.join(" "));

    // Open browser
    tracing::info!("Opening browser for OAuth login...");
    if let Err(e) = open::that(auth_url.as_str()) {
        tracing::warn!("Failed to open browser: {}. Please open this URL manually:\n{}", e, auth_url);
    }

    // Wait for the callback (with a timeout)
    let auth_code = wait_for_callback(listener).await?;

    // Exchange code for tokens
    let tokens = exchange_code(config, &auth_code, &verifier, &redirect_uri).await?;

    Ok(tokens)
}

/// Wait for the OAuth callback on the local server. Returns the auth code.
async fn wait_for_callback(listener: tokio::net::TcpListener) -> Result<String> {
    use tokio::time::{timeout, Duration};

    let result = timeout(Duration::from_secs(120), async {
        let (stream, _) = listener.accept().await?;
        let io = hyper_util::rt::TokioIo::new(stream);

        let (auth_code_tx, mut auth_code_rx) = tokio::sync::mpsc::channel::<String>(1);

        let service = hyper::service::service_fn(move |req: hyper::Request<hyper::body::Incoming>| {
            let tx = auth_code_tx.clone();
            async move {
                let uri = req.uri();
                if let Some(query) = uri.query() {
                    let params: Vec<(String, String)> = url::form_urlencoded::parse(query.as_bytes())
                        .map(|(k, v)| (k.to_string(), v.to_string()))
                        .collect();
                    if let Some((_, code)) = params.iter().find(|(k, _)| k == "code") {
                        let _ = tx.send(code.clone()).await;
                    }
                }
                let body = "Login successful! You can close this tab.";
                Ok::<_, hyper::Error>(hyper::Response::new(
                    http_body_util::Full::new(hyper::body::Bytes::from(body))
                ))
            }
        });

        hyper::server::conn::http1::Builder::new()
            .serve_connection(io, service)
            .await?;

        auth_code_rx.recv().await
            .ok_or_else(|| anyhow::anyhow!("no auth code received"))
    }).await;

    match result {
        Ok(Ok(code)) => Ok(code),
        Ok(Err(e)) => Err(e),
        Err(_) => Err(anyhow::anyhow!("OAuth login timed out after 120 seconds")),
    }
}

/// Exchange the authorization code for access + refresh tokens.
async fn exchange_code(
    config: &OAuthConfig,
    code: &str,
    verifier: &str,
    redirect_uri: &str,
) -> Result<OAuthTokens> {
    let client = reqwest::Client::new();

    let params = [
        ("grant_type", "authorization_code"),
        ("code", code),
        ("redirect_uri", redirect_uri),
        ("client_id", &config.client_id),
        ("code_verifier", verifier),
    ];

    let resp = client.post(&config.token_url)
        .form(&params)
        .send()
        .await?;

    if !resp.status().is_success() {
        let body = resp.text().await?;
        anyhow::bail!("token exchange failed: {}", body);
    }

    #[derive(serde::Deserialize)]
    struct TokenResponse {
        access_token: String,
        refresh_token: Option<String>,
        expires_in: Option<u64>,
    }

    let token_resp: TokenResponse = resp.json().await?;

    Ok(OAuthTokens {
        access_token: token_resp.access_token,
        refresh_token: token_resp.refresh_token,
        expires_in: token_resp.expires_in,
    })
}

/// Check if the token for a provider is expired or near-expiry.
/// If so, attempt to refresh it using the refresh_token.
/// Returns the valid (possibly refreshed) token string, or None if unavailable.
pub async fn refresh_if_needed(
    pool: &DbPool,
    store: &dyn SecretStore,
    provider: &str,
    token_url: &str,
    client_id: &str,
) -> Result<Option<String>> {
    let token = match get_token(pool, store, provider).await? {
        Some(t) => t,
        None => return Ok(None),
    };

    // Check expiry
    if let Some(ref expires_at) = token.expires_at {
        // Parse ISO 8601 datetime
        if let Ok(expiry) = chrono::DateTime::parse_from_rfc3339(expires_at) {
            let now = chrono::Utc::now();
            let buffer = chrono::Duration::minutes(5);
            if expiry - buffer > now {
                // Token still valid
                return Ok(Some(token.token));
            }
        }
    } else {
        // No expiry set — assume valid (API keys don't expire)
        return Ok(Some(token.token));
    }

    // Token expired or near-expiry — try to refresh
    let refresh_token = match token.refresh_token {
        Some(rt) => rt,
        None => {
            debug!(provider, "token expired and no refresh token available");
            return Ok(None);
        }
    };

    debug!(provider, "refreshing expired token");

    let client = reqwest::Client::new();
    let params = [
        ("grant_type", "refresh_token"),
        ("refresh_token", &refresh_token),
        ("client_id", client_id),
    ];

    let resp = client.post(token_url).form(&params).send().await?;
    if !resp.status().is_success() {
        let body = resp.text().await?;
        anyhow::bail!("token refresh failed for {}: {}", provider, body);
    }

    #[derive(serde::Deserialize)]
    struct RefreshResponse {
        access_token: String,
        refresh_token: Option<String>,
        expires_in: Option<u64>,
    }

    let refresh_resp: RefreshResponse = resp.json().await?;

    let new_expires_at = refresh_resp.expires_in.map(|secs| {
        let expiry = chrono::Utc::now() + chrono::Duration::seconds(secs as i64);
        expiry.to_rfc3339()
    });

    save_token(
        pool,
        store,
        provider,
        &refresh_resp.access_token,
        refresh_resp.refresh_token.as_deref().or(Some(&refresh_token)),
        new_expires_at.as_deref(),
    )
    .await?;

    Ok(Some(refresh_resp.access_token))
}

/// Resolve the API key / access token for a provider.
/// Checks in order: environment variable, then OS keychain.
/// Returns the token string or an error with login instructions.
pub async fn resolve_auth(
    pool: &DbPool,
    store: &dyn SecretStore,
    provider: &str,
    env_var: &str,
) -> Result<String> {
    // 1. Check environment variable
    if let Ok(key) = std::env::var(env_var) {
        if !key.is_empty() {
            return Ok(key);
        }
    }

    // 2. Check keychain
    if let Some(token) = get_token(pool, store, provider).await? {
        return Ok(token.token);
    }

    // 3. Not found
    anyhow::bail!(
        "No credentials found for {}. Set {} environment variable or run: kodo --login {}",
        provider, env_var, provider
    )
}
