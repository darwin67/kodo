* Motivation

All settings are currently hardcoded or passed via CLI args/env vars. Users need
a config file to persist preferences: default provider, model, theme, keybinds,
MCP servers, custom formatters, custom LSP servers, etc.

* Scope

Add config file loading from ~$HOME/.config/kodo/config.toml~ (global) and
~./kodo.toml~ (project-level, overrides global). Use TOML format — it's
the Rust ecosystem standard and human-friendly.

/// Top-level Kodo configuration.
#[derive(Debug, Clone, Deserialize, Default)]
#[serde(default)]
pub struct KodoConfig {
    /// Default LLM provider.
    pub provider: Option<String>,
    /// Default model name.
    pub model: Option<String>,
    /// Max tokens for LLM responses.
    pub max_tokens: Option<u32>,
    /// Default mode: "plan" or "build".
    pub mode: Option<String>,
    /// Theme: "dark" or "light".
    pub theme: Option<String>,
    /// MCP server configurations.
    pub mcp: Option<McpConfig>,
    /// Custom LSP server configurations.
    pub lsp: Option<LspConfig>,
    /// Custom formatter configurations.
    pub formatters: Option<FormattersConfig>,
    /// Keybind overrides.
    pub keybinds: Option<HashMap<String, String>>,
}

/// MCP server configuration section.
#[derive(Debug, Clone, Deserialize, Default)]
#[serde(default)]
pub struct McpConfig {
    /// Map of server name -> server config.
    pub servers: HashMap<String, McpServerEntry>,
}

/// A single MCP server entry.
#[derive(Debug, Clone, Deserialize)]
pub struct McpServerEntry {
    /// Command to run the server.
    pub command: String,
    /// Arguments to pass to the command.
    #[serde(default)]
    pub args: Vec<String>,
    /// Environment variables to set.
    #[serde(default)]
    pub env: HashMap<String, String>,
}

/// LSP server configuration section.
#[derive(Debug, Clone, Deserialize, Default)]
#[serde(default)]
pub struct LspConfig {
    /// Map of server name -> server config.
    pub servers: HashMap<String, LspServerEntry>,
}

/// A single custom LSP server entry.
#[derive(Debug, Clone, Deserialize)]
pub struct LspServerEntry {
    /// Command to run the server.
    pub command: String,
    /// Arguments to pass to the command.
    #[serde(default)]
    pub args: Vec<String>,
    /// File extensions this server handles (without dot).
    pub extensions: Vec<String>,
}

/// Formatter configuration section.
#[derive(Debug, Clone, Deserialize, Default)]
#[serde(default)]
pub struct FormattersConfig {
    /// Map of formatter name -> formatter config.
    pub formatters: HashMap<String, FormatterEntry>,
}

/// A single custom formatter entry.
#[derive(Debug, Clone, Deserialize)]
pub struct FormatterEntry {
    /// Command to run. Use $FILE for the file path placeholder.
    pub command: Vec<String>,
    /// File extensions this formatter handles (without dot).
    pub extensions: Vec<String>,
}

#+begin_src rust
impl KodoConfig {
    /// Load configuration from global and project-level config files.
    ///
    /// Resolution order (later overrides earlier):
    /// 1. Built-in defaults (all None)
    /// 2. Global: ~/.config/kodo/config.toml
    /// 3. Project: ./kodo.toml (in current directory)
    ///
    /// CLI args override config file values (handled by the caller).
    pub fn load() -> Result<Self> {
        let mut config = Self::default();

        // 1. Load global config
        if let Some(global_path) = global_config_path() {
            if global_path.exists() {
                debug!(path = %global_path.display(), "loading global config");
                let global = load_file(&global_path)?;
                config = merge(config, global);
            }
        }

        // 2. Load project config
        let project_path = PathBuf::from("kodo.toml");
        if project_path.exists() {
            debug!(path = %project_path.display(), "loading project config");
            let project = load_file(&project_path)?;
            config = merge(config, project);
        }

        Ok(config)
    }
}

/// Get the global config file path: ~/.config/kodo/config.toml
fn global_config_path() -> Option<PathBuf> {
    dirs::config_dir().map(|d| d.join("kodo").join("config.toml"))
}

/// Load and parse a single config file.
fn load_file(path: &Path) -> Result<KodoConfig> {
    let content = std::fs::read_to_string(path)?;
    let config: KodoConfig = toml::from_str(&content)?;
    Ok(config)
}

/// Merge two configs. Values from `override_cfg` take precedence over `base`.
fn merge(base: KodoConfig, override_cfg: KodoConfig) -> KodoConfig {
    KodoConfig {
        provider: override_cfg.provider.or(base.provider),
        model: override_cfg.model.or(base.model),
        max_tokens: override_cfg.max_tokens.or(base.max_tokens),
        mode: override_cfg.mode.or(base.mode),
        theme: override_cfg.theme.or(base.theme),
        mcp: override_cfg.mcp.or(base.mcp),
        lsp: override_cfg.lsp.or(base.lsp),
        formatters: override_cfg.formatters.or(base.formatters),
        keybinds: match (base.keybinds, override_cfg.keybinds) {
            (Some(mut base_kb), Some(override_kb)) => {
                base_kb.extend(override_kb);
                Some(base_kb)
            }
            (base_kb, override_kb) => override_kb.or(base_kb),
        },
    }
}

    fn parse_full_toml() {
        let toml_str = r#"
provider = "anthropic"
model = "claude-sonnet-4-20250514"
max_tokens = 4096
mode = "build"
theme = "dark"

[mcp.servers.filesystem]
command = "mcp-filesystem"
args = ["--root", "/home/user"]

[lsp.servers.clangd]
command = "clangd"
args = ["--background-index"]
extensions = ["c", "cpp", "h"]

[formatters.formatters.black]
command = ["black", "$FILE"]
extensions = ["py"]

[keybinds]
"ctrl+p" = "open_palette"
"ctrl+b" = "toggle_mode"
"#;
        let config: KodoConfig = toml::from_str(toml_str).unwrap();
        assert_eq!(config.provider, Some("anthropic".into()));
        assert_eq!(config.max_tokens, Some(4096));

        let mcp = config.mcp.unwrap();
        assert!(mcp.servers.contains_key("filesystem"));
        assert_eq!(mcp.servers["filesystem"].command, "mcp-filesystem");

        let lsp = config.lsp.unwrap();
        assert!(lsp.servers.contains_key("clangd"));
        assert_eq!(lsp.servers["clangd"].extensions, vec!["c", "cpp", "h"]);

        let fmt = config.formatters.unwrap();
        assert!(fmt.formatters.contains_key("black"));

        let kb = config.keybinds.unwrap();
        assert_eq!(kb["ctrl+p"], "open_palette");
    }

    fn merge_prefers_override() {
        let base = KodoConfig {
            provider: Some("anthropic".into()),
            model: Some("old-model".into()),
            ..Default::default()

        let override_cfg = KodoConfig {
            model: Some("new-model".into()),
            ..Default::default()
