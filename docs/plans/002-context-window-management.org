* Motivation

The agent accumulates all messages in ~Agent.messages~ (~Vec<Message>~, agent.rs:87)
with no awareness of context limits. ~ContextTracker~ (context.rs) is a simple
counter that never acts on its data. When a long session exceeds the model's
context window, the API returns an error and the session dies.

* Scope

Add token estimation per message, budget tracking against the model's context
window, and auto-compaction (summarization) when nearing the limit.

#+begin_src rust
/// Tracks token usage and manages context window budget.
#[derive(Debug)]
pub struct ContextTracker {
    pub total_input_tokens: u64,
    pub total_output_tokens: u64,
    /// The model's maximum context window size in tokens.
    context_window: u32,
    /// Tokens reserved for the model's response.
    max_output_tokens: u32,
    /// Estimated tokens used by the system prompt.
    system_prompt_tokens: u32,
    /// Fraction of budget at which to trigger compaction (0.0 - 1.0).
    compact_threshold: f32,
}

impl Default for ContextTracker {
    fn default() -> Self {
        Self {
            total_input_tokens: 0,
            total_output_tokens: 0,
            context_window: 200_000,    // sensible default
            max_output_tokens: 8192,
            system_prompt_tokens: 0,
            compact_threshold: 0.80,
        }
    }
}

impl ContextTracker {
    pub fn new() -> Self {
        Self::default()
    }

    /// Set context window parameters. Call after selecting a model.
    pub fn configure(&mut self, context_window: u32, max_output_tokens: u32, system_prompt: &str) {
        self.context_window = context_window;
        self.max_output_tokens = max_output_tokens;
        self.system_prompt_tokens = estimate_tokens(system_prompt);
    }

    /// Record token usage from a completion.
    pub fn record(&mut self, usage: &Usage) {
        self.total_input_tokens += usage.input_tokens as u64;
        self.total_output_tokens += usage.output_tokens as u64;
    }

    /// Total tokens consumed in this session.
    pub fn total_tokens(&self) -> u64 {
        self.total_input_tokens + self.total_output_tokens
    }

    /// The model's context window size.
    pub fn context_window(&self) -> u32 {
        self.context_window
    }

    /// Available budget for conversation messages (excluding system prompt and output reserve).
    pub fn budget(&self) -> u32 {
        self.context_window
            .saturating_sub(self.max_output_tokens)
            .saturating_sub(self.system_prompt_tokens)
    }

    /// Estimated tokens currently used by the given messages.
    pub fn messages_token_count(&self, messages: &[Message]) -> u32 {
        messages.iter().map(|m| m.estimated_tokens).sum()
    }

    /// True when messages exceed the compact threshold of the budget.
    pub fn should_compact(&self, messages: &[Message]) -> bool {
        let used = self.messages_token_count(messages);
        let threshold = (self.budget() as f32 * self.compact_threshold) as u32;
        used > threshold
    }
}

#+begin_src rust
/// Number of recent turns (user+assistant pairs) to keep during compaction.
const COMPACT_KEEP_TURNS: usize = 4;

/// Compact the conversation by summarizing old messages.
/// Keeps the most recent COMPACT_KEEP_TURNS turns intact.
async fn compact_context(&mut self, tx: Option<&AgentEventTx>) -> Result<()> {
    let keep_count = COMPACT_KEEP_TURNS * 2; // each turn = user + assistant
    if self.messages.len() <= keep_count {
        return Ok(()); // not enough messages to compact
    }

    let before_tokens = self.context.messages_token_count(&self.messages);

    let split_at = self.messages.len() - keep_count;
    let old_messages = &self.messages[..split_at];

    // Build a summary of old messages
    let mut summary_input = String::from(
        "Summarize the following conversation history concisely. \
         Preserve all file paths, function names, decisions made, \
         and tool results. This summary will replace the original \
         messages as context for continuing the conversation.\n\n"
    );
    for msg in old_messages {
        let role = match msg.role {
            Role::User => "User",
            Role::Assistant => "Assistant",
        };
        summary_input.push_str(&format!("[{}]: {}\n\n", role, msg.text()));
    }

    // Send summarization request to the LLM
    let summary_request = CompletionRequest {
        model: self.model.clone(),
        system: Some("You are a conversation summarizer. Be concise but preserve all technical details.".to_string()),
        messages: vec![Message::user(summary_input)],
        tools: vec![],
        max_tokens: 2048,
    };

    let response = self.provider.complete(summary_request).await?;
    let summary_text = response.message.text();
    self.context.record(&response.usage);

    // Replace old messages with a single summary message
    let recent_messages: Vec<Message> = self.messages[split_at..].to_vec();
    self.messages.clear();
    self.messages.push(Message::user(format!(
        "[Conversation summary (compacted)]\n{}",
        summary_text
    )));
    self.messages.extend(recent_messages);

    let after_tokens = self.context.messages_token_count(&self.messages);

    emit(tx, AgentEvent::ContextCompacted {
        before_tokens,
        after_tokens,
    });

    debug!(
        before = before_tokens,
        after = after_tokens,
        "context compacted"
    );

    Ok(())
}

/// Look up the context window size for a known model.

fn context_window_for_model(model: &str) -> u32 {
    match model {
        m if m.contains("opus") => 200_000,
        m if m.contains("sonnet") => 200_000,
        m if m.contains("haiku") => 200_000,
        m if m.contains("gpt-4o") => 128_000,
        m if m.contains("gpt-4") => 128_000,
        m if m.contains("gpt-3.5") => 16_385,
        m if m.contains("gemini") => 1_000_000,
        m if m.contains("llama") => 128_000,
        _ => 128_000, // conservative default
    }
}
